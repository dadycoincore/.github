### API Integration

```javascript
// Install SDK
npm install @dadycoin/sdk

// Initialize
import DadyCoin from '@dadycoin/sdk';
const client = new DadyCoin({ apiKey: 'your_api_key' });

// Create payment
const invoice = await client.invoices.create({
  amount: 100,
  currency: 'USD',
  description: 'Order #123'
});
```
